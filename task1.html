Sub Button4_Click() Dim wsGL As Worksheet, wsExp As Worksheet Dim lastRowGL As Long, lastRowExp As Long Dim i As Long, j As Long Dim countryGL As String, countryExp As String Dim ccValue As Variant, sumW As Double Dim dictD As Object, dictF As Object, dictH As Object, dictI As Object ' Set worksheet references On Error Resume Next Set wsGL = ThisWorkbook.Worksheets("GL") Set wsExp = ThisWorkbook.Worksheets("Exposures by geography") On Error GoTo 0 If wsGL Is Nothing Then MsgBox "GL sheet not found. Please import data first.", vbCritical Exit Sub End If If wsExp Is Nothing Then MsgBox "Exposures by geography sheet not found.", vbCritical Exit Sub End If ' Find the last row with data in GL sheet (using Column J as anchor) lastRowGL = wsGL.Cells(wsGL.Rows.Count, "J").End(xlUp).Row If lastRowGL < 2 Then MsgBox "No data found in GL sheet to process.", vbExclamation Exit Sub End If ' Find the last row in Exposures by geography sheet (using Column C) lastRowExp = wsExp.Cells(wsExp.Rows.Count, "C").End(xlUp).Row If lastRowExp < 6 Then lastRowExp = 6 ' Ensures we cover up to row 6 at least ' Initialize Dictionaries to aggregate Sum(W) grouped by Country Set dictD = CreateObject("Scripting.Dictionary") Set dictF = CreateObject("Scripting.Dictionary") Set dictH = CreateObject("Scripting.Dictionary") Set dictI = CreateObject("Scripting.Dictionary") ' Loop through GL data starting from Row 2 (assuming Row 1 has headers) For i = 2 To lastRowGL ccValue = wsGL.Cells(i, "J").Value countryGL = Trim(CStr(wsGL.Cells(i, "H").Value)) sumW = Val(wsGL.Cells(i, "W").Value) ' Handle blank/null countries safely If countryGL = "" Then countryGL = "BLANK_COUNTRY" ' Evaluate Cases based on Column J value Select Case ccValue Case 100, 361 ' Case 1 dictD(countryGL) = dictD(countryGL) + sumW Case 364 ' Case 2 dictF(countryGL) = dictF(countryGL) + sumW Case 109 ' Case 3 dictH(countryGL) = dictH(countryGL) + sumW Case 300, 380 ' Case 4 dictI(countryGL) = dictI(countryGL) + sumW End Select Next i ' Populate data into "Exposures by geography" sheet starting from Row 6 Application.ScreenUpdating = False For j = 6 To lastRowExp countryExp = Trim(CStr(wsExp.Cells(j, "C").Value)) If countryExp = "" Then countryExp = "BLANK_COUNTRY" ' Column D (Case 1) If dictD.Exists(countryExp) Then wsExp.Cells(j, "D").Value = dictD(countryExp) Else wsExp.Cells(j, "D").Value = 0 End If ' Column F (Case 2) If dictF.Exists(countryExp) Then wsExp.Cells(j, "F").Value = dictF(countryExp) Else wsExp.Cells(j, "F").Value = 0 End If ' Column H (Case 3) If dictH.Exists(countryExp) Then wsExp.Cells(j, "H").Value = dictH(countryExp) Else wsExp.Cells(j, "H").Value = 0 End If ' Column I (Case 4) If dictI.Exists(countryExp) Then wsExp.Cells(j, "I").Value = dictI(countryExp) Else wsExp.Cells(j, "I").Value = 0 End If Next j Application.ScreenUpdating = True MsgBox "Exposures calculated and updated successfully!", vbInformation End Sub
