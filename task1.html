Sub PopulateExposuresByGeography() Dim glWs As Worksheet, expWs As Worksheet Dim lastRowGL As Long, lastRowExp As Long Dim i As Long, j As Long Dim ccDesc As String, ccCode As Variant, country As String Dim amt As Double Dim destCol As String Dim matchFound As Boolean Set glWs = ThisWorkbook.Worksheets("GL") Set expWs = ThisWorkbook.Worksheets("Exposures by geography") lastRowGL = glWs.Cells(glWs.Rows.Count, "K").End(xlUp).Row lastRowExp = expWs.Cells(expWs.Rows.Count, "C").End(xlUp).Row ' Clear previous results in D, F, H, I (rows 2 to lastRowExp) expWs.Range("D2:D" & lastRowExp).ClearContents expWs.Range("F2:F" & lastRowExp).ClearContents expWs.Range("H2:H" & lastRowExp).ClearContents expWs.Range("I2:I" & lastRowExp).ClearContents ' Loop through each country row in Exposures sheet For j = 2 To lastRowExp Dim countryExp As String countryExp = Trim(expWs.Cells(j, "C").Value) Dim sumCase1 As Double, sumCase2 As Double, sumCase3 As Double, sumCase4 As Double sumCase1 = 0: sumCase2 = 0: sumCase3 = 0: sumCase4 = 0 ' Loop through GL rows once per country (could be optimized with Dictionary, see note below) For i = 2 To lastRowGL ccDesc = Trim(UCase(glWs.Cells(i, "K").Value)) ccCode = glWs.Cells(i, "J").Value country = Trim(glWs.Cells(i, "H").Value) amt = glWs.Cells(i, "W").Value If country = countryExp And Len(country) > 0 Then ' Case 1: CENTRAL BANK, CENTRAL ADMINISTR. + J = 100 or 361 If (ccDesc = "CENTRAL BANK" Or ccDesc = "CENTRAL ADMINISTR.") And _ (ccCode = 100 Or ccCode = 361) Then sumCase1 = sumCase1 + amt End If ' Case 2: PUBLIC COMPANY + J = 364 If ccDesc = "PUBLIC COMPANY" And ccCode = 364 Then sumCase2 = sumCase2 + amt End If ' Case 3: BANK + J = 109 If ccDesc = "BANK" And ccCode = 109 Then sumCase3 = sumCase3 + amt End If ' Case 4: CORPORATE, PRIVATE ORGANIZATION + J = 300 or 380 If (ccDesc = "CORPORATE" Or ccDesc = "PRIVATE ORGANIZATION") And _ (ccCode = 300 Or ccCode = 380) Then sumCase4 = sumCase4 + amt End If End If Next i ' Populate results (0 if blank/no match — already 0 by default) expWs.Cells(j, "D").Value = sumCase1 expWs.Cells(j, "F").Value = sumCase2 expWs.Cells(j, "H").Value = sumCase3 expWs.Cells(j, "I").Value = sumCase4 Next j MsgBox "Exposures by geography populated successfully.", vbInformation End Sub
